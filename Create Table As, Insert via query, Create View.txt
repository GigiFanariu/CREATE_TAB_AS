CREATE VIEW EXPENSIVE_MEAL AS SELECT * FROM MEAL 
WHERE PRICE > 15;

CREATE VIEW LACTOSE_FREE_INGREDIENT AS SELECT * FROM INGREDIENT
WHERE !CONTAINS_LACTOSE;
